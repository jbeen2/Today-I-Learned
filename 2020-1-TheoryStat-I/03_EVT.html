function1<-function(aa) {
  mylm1<-lm(log(1+X/aa)~probit,data=kospi2)
  slope1<-as.numeric(format(summary(mylm1)$coefficients[2],digits=2))
  r2<-summary(mylm1)$r.squared
  return(list(r2,slope1))
}

interval<-0.01
a<-seq(interval,0.5,by=interval)
Rsquared1<-rep(NA,length(a))

for (i in 1:length(a)) {
  Rsquared1[i]<-function1(a[i])[[1]]
}

a_final1<-interval*which.max(Rsquared1)
gamma_final1<-function1(a_final1)[[2]]
sigma_final1<-a_final1*gamma_final1

ggplot(kospi2,aes(probit,log(1+X/a_final1)))+geom_point()+geom_smooth(method="lm")+ggtitle(paste("a=",a_final1,",","R2=",format(Rsquared1[which.max(Rsquared1)],digits=3),",","gamma=",gamma_final1,",","sigma=",sigma_final1))

function2<-function(g) {
  kospi3 <-kospi2 %>% mutate(r=rank(X),probit=((1-r/(nrow(kospi2)+1))^(-g)-1)/g) %>% arrange(X)
  mylm2<-lm(X~probit,data=kospi3)
  slope2<-as.numeric(format(summary(mylm2)$coefficients[2],digits=2))
  r2_2=summary(mylm2)$r.squared
  return(list(r2_2,slope2))
}

initial<-0.01
gg<-seq(initial,1,by=initial)
Rsquared2<-rep(NA,length(gg))

for (j in 1:length(gg)) {
  Rsquared2[j]<-function2(gg[j])[[1]]
}

gamma_final2<-which.max(Rsquared2)*initial
sigma_final2<-function2(gamma_final2)[[2]]

kospi3<-kospi2 %>% mutate(r=rank(X),probit=((1-r/(nrow(kospi2)+1))^(-gamma_final2)-1)/gamma_final2) %>% arrange(X)

ggplot(kospi3,aes(probit,X))+geom_point()+geom_smooth(method="lm")+ggtitle(paste("R2=",format(Rsquared2[which.max(Rsquared2)],digits=3),",","gamma=",gamma_final2,",","sigma=",sigma_final2))
