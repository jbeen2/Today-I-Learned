Part1. Smoking and Mortality: 흡연과 수명의 연관성 분석¶

  • Data: Canadian Veteran Survey
  • Objective: NonSmoker와 Cigarette Smoker의 생명표(life table)를 작성하여 차이 비교

(a) NonSmoker & Cigarette 그룹의 사망확률 산점도¶

In [4]:
smoker <- read.csv("smoker.csv", header=T)
colnames(smoker) <- c("age","x","pop.non","death.non","pop.cig","death.cig")

# tqx: 사망자수 / 인구수 
non<-smoker %>% transmute(x,pop=pop.non,death=death.non,tqx=death.non/pop.non,ln=log(-log(1-tqx)))
cigar<-smoker %>% transmute(x,pop=pop.cig,death=death.cig,tqx=death.cig/pop.cig,ln=log(-log(1-tqx)))

smoking<-rbind(non,cigar)
smoking$smoke<-c(rep("nonsmoking",9),rep("smoking",9))

  • 산점도를 보았을 때 나이가 많아질수록 사망확률이 크게 나타났다.
  • 또한 비흡연자와 흡연자의 그룹을 비교했을 때 흡연자의 사망확률이 더 높았으나,
    나이가 많아질수록 비흡연자와의 차이가 더 크게 나타나고 있다.
  • (x, tqx)의 산점도를 보았을 때 지수형태로 따르는 것을 보이고,
  • (x,ln[-ln(1-tqx)])으로 보았을 때 선형을 따르고 있다.
  • 따라서 Gompertz 법칙이 성립함을 알 수 있다.

#MLE
non_mle<-function(v) {
  z=v[1]+v[2]*non$x
  Gz=1-exp(-exp(z))
  sum(log(dbinom(non$death,non$pop,Gz)))
}

# 3-parameter Weibull
mylm<-function(a) {
  locomotive<-locomotive %>% mutate(weibull3=log(y-a))
  lm2<-lm(weibull3~gompit,data=locomotive)
  intercept<-as.numeric(format(summary(lm2)$coefficients[1],digits=3))
  slope<-as.numeric(format(summary(lm2)$coefficients[2],digits=3))
  r2<-summary(lm2)$r.squared
  return(list(r2,intercept,slope))
}

a<-seq(0,10000,by=100)

  Rsquared[i]<-mylm(a[i])[[1]]

# 2-parameter Weibull
weibull2<-function(v) {
  u<-(locomotive$lny-v[1])/v[2]
  g0<-exp(u-exp(u))
  G0<-1-exp(-exp(u))
  sum(log((g0/v[2]/locomotive$y)^locomotive$tag*(1-G0)^(1-locomotive$tag)))
}

# 3-parameter Weibull
weibull3<-function(v) {
  locomotive<-locomotive %>% mutate(weibullq=log(y-v[3]))
  u<-(locomotive$weibullq-v[1])/v[2]
  g0<-exp(u-exp(u))
  G0<-1-exp(-exp(u))
  sum(log((g0/(v[2]*(locomotive$y-v[3])))^locomotive$tag*(1-G0)^(1-locomotive$tag)))
}

# log-normal
lognormal<-function(v) {
  u<-(locomotive$lny-v[1])/v[2]
  g0<-dnorm(u)
  G0<-pnorm(u)
  sum(log((g0/v[2]/locomotive$y)^locomotive$tag*(1-G0)^(1-locomotive$tag)))
}

# Gompertz
gompertz<-function(v) {
  u<-(locomotive$y-v[1])/v[2]
  g0<-exp(u-exp(u))
  G0<-1-exp(-exp(u))
  sum(log((g0/v[2])^locomotive$tag*(1-G0)^(1-locomotive$tag)))
}

# Gumbel
gumbel<-function(v) {
  u<-(locomotive$y-v[1])/v[2]
  g0<-exp(-u-exp(-u))
  G0<-exp(-exp(-u))
  sum(log((g0/v[2])^locomotive$tag*(1-G0)^(1-locomotive$tag)))
}

# Normal

  u<-(locomotive$y-v[1])/v[2]
  g0<-dnorm(u)
  G0<-pnorm(u)
  sum(log((g0/v[2])^locomotive$tag*(1-G0)^(1-locomotive$tag)))
}
